求解下列微分方程,并指出这些方程在Oxy平面上有意义的区域.
\begin{exercise}[2-2,1,(1)]
$$
\frac{dy}{dx}=\frac{x^2}{y}.
$$
\end{exercise}
\begin{proof}[解]
  将该方程化为
  \begin{equation}
    \label{eq:12.48pm}
    ydy-x^2dx=0,
  \end{equation}
其中 $y\neq 0$.式~(\ref{eq:12.48pm})是一个恰当微分方程,因此我们可以求
通积分,设
\begin{equation}
  \label{eq:13.16pm}
\frac{\pa \phi}{\pa y}=y,
\end{equation}
于是,
\begin{equation}
  \label{eq:13.20}
  \phi=\frac{1}{2}y^2+f(x).
\end{equation}
代入下式,
\begin{equation}
  \label{eq:13.19pm}
  \frac{\pa \phi}{\pa x}=-x^2
\end{equation}
可得
\begin{equation}
  f'(x)=x^2\Rightarrow f(x)=-\frac{1}{3}x^3+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:13.25}
  \frac{1}{2}y^2-\frac{1}{3}x^3+C=0.
\end{equation}
有意义的区域为 $\mathbf{R}^2$ 上纵坐标不为0的点的集合.
\end{proof}
\begin{exercise}[2-2,1,(2)]
$$
\frac{dy}{dx}=\frac{x^2}{y(1+x^3)}.
$$
\end{exercise}
\begin{proof}[解]
  通过分离变量,得到
  \begin{equation}
    \label{eq:13.49pm}
    ydy=\frac{x^2}{1+x^3}dx.
  \end{equation}
其中 $y(1+x^3)\neq 0$.易得 \eqref{eq:13.49pm}是一个恰当方程,因此可以求
通积分.设
\begin{equation}
  \label{eq:13.51pm}
  \frac{\pa \phi}{\pa y}=y,
\end{equation}
可得
\begin{equation}
  \label{eq:13.52}
  \phi=\frac{1}{2}y^2+f(x).
\end{equation}
代入式子
\begin{equation}
  \label{eq:13.53}
  \frac{\pa\phi}{\pa x}=-\frac{x^2}{1+x^3},
\end{equation}
可得
\begin{equation}
  \label{eq:14.04}
  f'(x)=-\frac{x^2}{1+x^3}\Rightarrow f(x)=-\frac{1}{3}\ln |(1+x^3)|+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:14.13}
  \frac{1}{2}y^2-\frac{1}{3}\ln|(1+x^3)|+C=0.
\end{equation}
其中区域是 $\mathbf{R}^2\backslash\{(x,y):y(1+x^3)=0.\}$.
\end{proof}
\begin{exercise}[2-2,1,(3)]
$$
\frac{dy}{dx}+y^2\sin x=0.
$$  
\end{exercise}
\begin{proof}[解]
  可得
$$
\frac{dy}{dx}=-y^2\sin x,
$$
当 $y\neq 0$ 时,也即
\begin{equation}
  \label{eq:14.31pm}
  \frac{1}{y^2}dy=-\sin xdx.
\end{equation}
于是
\begin{equation}
  \label{eq:14.42}
  \frac{\phi}{\pa y}=\frac{1}{y^2}\Rightarrow \phi=-\frac{1}{y}+f(x).
\end{equation}
于是
\begin{equation}
  \label{eq:14.43}
  f'(x)=\sin x\ri f(x)=-\cos x+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:14.45}
  -\frac{1}{y}-\cos x+C=0.
\end{equation}
当 $y=0$ 时,可得
\begin{equation}
  \label{eq:14.47pm}
  \frac{dy}{dx}=0\ri y=C.
\end{equation}
可见,区域是整个 $\mathbf{R}^2$.
\end{proof}
\begin{exercise}[2-2,1,(4)]
$$
\frac{dy}{dx}=1+x+y^2+xy^2.
$$  
\end{exercise}
\begin{proof}[解]
可得
\begin{equation}
  \label{eq:16.32pm}
  \frac{dy}{dx}=(1+y^2)(1+x).
\end{equation}
因此
\begin{equation}
  \label{eq:16.34pm}
  \frac{1}{1+y^2}dy=(1+x)dx.
\end{equation}
易得~\eqref{eq:16.34pm}是一个恰当方程.因此可以求通积分,设
\begin{equation}
  \label{eq:16.37pm}
  \frac{\pa\phi}{\pa y}=\frac{1}{1+y^2},
\end{equation}
则
\begin{equation}
  \label{eq:16.38pm}
  \phi=\arctan y+f(x).
\end{equation}
于是
\begin{equation}
  \label{eq:16.40}
  f'(x)=-1-x\ri f(x)=-\frac{1}{2}x^2-x+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:16.46}
  \arctan y-\frac{1}{2}x^2-x+C=0.
\end{equation}
\end{proof}
\begin{exercise}[2-2,1,(5)]
$$
  \frac{dy}{dx}=(\cos x\cos 2y)^2.
$$
\end{exercise}
\begin{proof}[解]
  可得当 $\cos2y\neq 0$ 时,
  \begin{equation}
    \label{eq:16.58}
    \frac{1}{(\cos 2y)^2}dy=\cos^2x dx.
  \end{equation}
也即
\begin{equation}
  \label{eq:17.00pm}
(1+\tan^{2} 2y)dy-(1+\tan^2x)dx=0.
\end{equation}
因此
\begin{equation}
  \label{eq:17.16}
  \frac{\pa \phi}{\pa x}=-(1+\tan^2x)\ri \phi=-\tan x+f(y).
\end{equation}
于是
\begin{equation}
  \label{eq:17.18}
  f'(y)=1+\tan^22y\ri f(y)=\frac{1}{2}\tan 2y+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:17.20}
  -\tan x+\frac{1}{2}\tan2y+C=0.
\end{equation}
当 $\cos2y=0$ 时,可得 $y=C$.可见微分方程是定义在整个 $\mathbf{R}^2$ 上
的.
\end{proof}
\begin{exercise}[2-2,1,(6)]
$$
x \frac{dy}{dx}=\sqrt{1-y^2}.
$$
\end{exercise}
\begin{proof}[解]
当 $y\in (-1,1)$,$x\neq 0$ 时,可得恰当方程
  \begin{equation}
    \label{eq:17.36}
    \frac{1}{\sqrt{1-y^2}}dy-\frac{1}{x}dx=0.
  \end{equation}
于是可得 $\phi=-\ln |x|+f(y)$,于是
\begin{equation}
  \label{eq:17.39}
  f'(y)=\frac{1}{\sqrt{1-y^2}}. 
\end{equation}
可得 $f(y)=\sin^{-1}y+C$.因此通积分为
\begin{equation}
  \label{eq:20.48pm}
  -\ln|x|+\sin^{-1}y+C=0.
\end{equation}
当 $x=0$ 时,$y=\pm 1$.当 $x\neq 0$, $y=\pm 1$ 时,
\begin{equation}
  \label{eq:21.11pm}
  x \frac{dy}{dx}=0.
\end{equation}
于是可得 $y=C$.
\end{proof}
\begin{exercise}[2-2,1,(7)]
$$
\frac{dy}{dx}=\frac{x-e^{-x}}{y+e^y}.
$$  
\end{exercise}
\begin{proof}[解]
我们要求 $y+e^y\neq 0$.可得恰当微分方程
  \begin{equation}
    \label{eq:21.17}
    (y+e^y)dy-(x-e^{-x})dx=0.
  \end{equation}
于是
\begin{equation}
  \label{eq:21.19}
  \frac{\pa \phi}{\pa y}=y+e^y,
\end{equation}
因此
\begin{equation}
  \label{eq:21.20pm}
  \phi=\frac{1}{2}y^2+e^y+f(x).
\end{equation}
因此
\begin{equation}
  \label{eq:21.21pm}
  f'(x)=-x+e^{-x}\ri f(x)=\frac{-1}{2}x^2-e^{-x}+C.
\end{equation}
因此通积分为
\begin{equation}
  \label{eq:21.22pm}
  \frac{1}{2}y^2+e^y-\frac{1}{2}x^2-e^{-x}+C=0.
\end{equation}
\end{proof}
