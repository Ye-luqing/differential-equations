  \begin{theorem}[7.2]
    Let the domain $D$ be convex and the function $f(x,y)$ be
    differentiable with respect to $y$ in $D$.Then for the Lipschitz
    condition
$$
|f(x,y_1)-f(x,y_2)|\leq L|y_1-y_2|
$$
to be satisfied,it is necessary and sufficient that
$$
\sup_D \left|\frac{\pa f(x,y)}{\pa y}\right|\leq L.
$$
\end{theorem}
\begin{proof}
  $\ri:\forall (x,y)\in D$,
$$\left|\frac{\pa f(x,y)}{\pa y}\right|=\lim_{\Delta y\to 0;\Delta y\neq
  0}\left|\frac{f(x,y+\Delta y)-f(x,y)}{\Delta y}\right|\leq L.
$$
So we have
$$
\sup_D\left |\frac{\pa f(x,y)}{\pa y}\right|\leq L.
$$

$\Leftarrow:$This can be easily proved by the differential mean value theorem.
\end{proof}
