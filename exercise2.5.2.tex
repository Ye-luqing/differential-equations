  \begin{exercise}[2.5.2]
证明方程
\begin{equation}
  \label{eq:1}
  P(x,y)dx+Q(x,y)dy=0
\end{equation}
有形如 $u=u(\phi(x,y))$ 的积分因子的充要条件是
$$
\frac{\frac{\pa P}{\pa y}-\frac{\pa Q}{\pa x}}{Q \frac{\pa\phi}{\pa
    x}-P \frac{\pa\phi}{\pa y}}=f(\phi(x,y)),
$$
并写出这个积分因子.
  \end{exercise}
  \begin{proof}[证明]
由题目条件可知,
\begin{equation}
  \label{eq:2}
  u(\phi(x,y))P(x,y)dx+u(\phi(x,y))Q(x,y)dy=0
\end{equation}
是一个恰当微分方程,当且仅当
\begin{align*}
  \frac{\pa [u(\phi(x,y))P(x,y)]}{\pa y}=\frac{\pa
    [u(\phi(x,y))Q(x,y)]}{\pa x}.
\end{align*}
即
\begin{align*}
  \frac{\pa u(\phi(x,y))}{\pa y}P(x,y)+u(\phi(x,y))\frac{\pa P}{\pa
    y}=\frac{\pa u(\phi(x,y))}{\pa x}Q(x,y)+u(\phi(x,y))\frac{\pa
    Q}{\pa x}.
\end{align*}
根据链法则,即
\begin{align*}
  \frac{\pa u}{\pa \phi}\frac{\pa \phi}{\pa
    y}P(x,y)+u(\phi(x,y))\frac{\pa P}{\pa y}=\frac{\pa u}{\pa
    \phi}\frac{\pa\phi}{\pa x}Q(x,y)+u(\phi(x,y))\frac{\pa Q}{\pa x}.
\end{align*}
整理一下,可得
\begin{align*}
 u(\phi(x,y)) \frac{\frac{\pa P}{\pa y}-\frac{\pa Q}{\pa
     x}}{Q(x,y)\frac{\pa \phi}{\pa x}-P(x,y)\frac{\pa\phi}{\pa
     y}}=\frac{\pa u}{\pa\phi}.
\end{align*}

  \end{proof}
